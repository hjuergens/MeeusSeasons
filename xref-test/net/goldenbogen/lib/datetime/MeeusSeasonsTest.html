14      /**
15       * Equinoxes and Solstices for year 2016
16       * @see <a href="http://www.geoastro.de/astro/astroJS/seasons/index.htm">http://www.geoastro.de/astro/astroJS/seasons/index.htm</a>
17       * @throws Exception description
18       */
19      @Test
20      public void testGetNorthernHemisphereSeason2016() throws Exception {
21          final DateFormat formatter =
22                  new SimpleDateFormat("EEE, MMM d, ''yy 'at' H:mm z", Locale.ENGLISH);
23  
24          assertEquals(getNorthernHemisphereSeason(
25                  formatter.parse("Sun, Mar 20, '16 at 4:29 UTC")), 1);// : Winter
26  
27          assertEquals(getNorthernHemisphereSeason(
28                  formatter.parse("Sun, Mar 20, '16 at 4:30 UTC")), 2);// : Spring
29  
30          assertEquals(getNorthernHemisphereSeason(
31                  formatter.parse("Mon, Jun 20, '16 at 22:33 UTC")), 2);// : Spring
32  
33          assertEquals(getNorthernHemisphereSeason(
34                  formatter.parse("Mon, Jun 20, '16 at 22:34 UTC")), 3);// : Summer
35  
36          assertEquals(getNorthernHemisphereSeason(
37                  formatter.parse("Thu, Sep 22, '16 at 14:20 UTC")), 3);// : Summer
38  
39          assertEquals(getNorthernHemisphereSeason(
40                  formatter.parse("Thu, Sep 22, '16 at 14:21 UTC")), 4);// : Autumn
41  
42          assertEquals(getNorthernHemisphereSeason(
43                  formatter.parse("Wed, Dec 21, '16 at 10:43 UTC")), 4);// : Autumn
44  
45          assertEquals(getNorthernHemisphereSeason(
46                  formatter.parse("Wed, Dec 21, '16 at 10:44 UTC")), 4);// : Autumn !!!
47  
48          assertEquals(getNorthernHemisphereSeason(
49                  formatter.parse("Wed, Dec 21, '16 at 10:45 UTC")), 0);// : Winter
50      }
