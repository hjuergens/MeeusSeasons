19  	 * Returns the value of the current season.
20  	 * 
21  	 * @return int = 1: Winter<br>
22  	 *         int = 2: Spring<br>
23  	 *         int = 3: Summer<br>
24  	 *         int = 4: Autumn
25  	 * @throws Exception description
26  	 */
27  	public static int getActualNorthernHemisphereSeason() throws Exception {
28  		for (int iSeason = 1; iSeason <= 4; iSeason++) {
29  			if (calcEquSol(iSeason, Calendar.getInstance().get(Calendar.YEAR)).getTime() > new Date().getTime()) {
30  				return iSeason;
31  			}
32  		}
33  		return 0;
34  	}

36  	public static int getNorthernHemisphereSeason(Date date) throws Exception {
37  		final Calendar calendar = Calendar.getInstance();
38  		calendar.setTime(date);
39  		final int year = calendar.get(Calendar.YEAR);
40  		for (int iSeason = 1; iSeason <= 4; iSeason++) {
41  			if (calcEquSol(iSeason, year).getTime() > date.getTime()) {
42  				return iSeason;
43  			}
44  		}
45  		return 0;
46  	}
47  	/**
48  	 * Special implementation of the Math.cos function.
49  	 * 
50  	 * @param value a number
51  	 * @return double
52  	 */
53  	private static double specialCos(double value) {
54  		return Math.cos(value * Math.PI / 180);
55  	}

57  	/**
58  	 * Calculate a single event for a single year (Either a Equiniox or Solstice)<br>
59  	 * <i>See Meeus Astronmical Algorithms Chapter 27</i>
60  	 * 
61  	 * @param iSeason to calculate
62  	 * @param year for calculation
63  	 * @throws Exception
64  	 */
65  	private static Date calcEquSol(int iSeason, int year) throws Exception {
66  		int k = iSeason - 1;
67  		// Initial estimate of date of event
68  		double ETA = init(k, year);
69  		double T = (ETA - 2451545.0) / 36525;
70  		double W = 35999.373 * T - 2.47;
71  		double dL = 1.0 + 0.0334 * specialCos(W) + 0.0007 * specialCos(2 * W);
72  		double S = periodic24(T);
73  		// This is the answer in Julian Emphemeris Days
74  		double JDE = ETA + ((0.00001 * S) / dL);
75  		// Convert Julian Days to TDT in a Date Object
76  		Date TDT = fromJDEToUTC(JDE);
77  		// Correct TDT to UTC, both as Date Objects
78  		Date UTC = fromTDTToUTC(TDT);
79  		return UTC;
80  	}

82  	/**
83  	 * Calculate an initial guess as the ETA of the Equinox or Solstice of a given year.<br>
84  	 * <b>Valid for years 1000 to 3000!</b><br>
85  	 * <br>
86  	 * <i>See Meeus Astronmical Algorithms Chapter 27</i>
87  	 * 
88  	 * @param k = season
89  	 * @param year for calculation
90  	 * @return
91  	 */
92  	private static double init(int k, int year) {
93  		double tmpYear = ((double) year - 2000) / 1000;
94  		double result = 0.0;
95  		switch (k) {
96  			case 0:
97  				result = 2451623.80984 + 365242.37404 * tmpYear + 0.05169 * Math.pow(tmpYear, 2) - 0.00411 * Math.pow(tmpYear, 3) - 0.00057 * Math.pow(tmpYear, 4);
98  				break;
99  			case 1:
100 				result = 2451716.56767 + 365241.62603 * tmpYear + 0.00325 * Math.pow(tmpYear, 2) + 0.00888 * Math.pow(tmpYear, 3) - 0.00030 * Math.pow(tmpYear, 4);
101 				break;
102 			case 2:
103 				result = 2451810.21715 + 365242.01767 * tmpYear - 0.11575 * Math.pow(tmpYear, 2) + 0.00337 * Math.pow(tmpYear, 3) + 0.00078 * Math.pow(tmpYear, 4);
104 				break;
105 			case 3:
106 				result = 2451900.05952 + 365242.74049 * tmpYear - 0.06223 * Math.pow(tmpYear, 2) - 0.00823 * Math.pow(tmpYear, 3) + 0.00032 * Math.pow(tmpYear, 4);
107 				break;
108 			default:
109 				result = 0.0;
110 				break;
111 		}
112 		return result;
113 	}

115 	/**
116 	 * Calculate 24 Periodic Terms.<br>
117 	 * <br>
118 	 * <i>See Meeus Astronmical Algorithms Chapter 27</i>
119 	 * 
120 	 * @param T
121 	 * @return
122 	 */
123 	private static double periodic24(double T) {
124 		double result = 0.0;
125 		double[] A = { 485, 203, 199, 182, 156, 136, 77, 74, 70, 58, 52, 50, 45, 44, 29, 18, 17, 16, 14, 12, 12, 12, 9, 8 };
126 		double[] B = { 324.96, 337.23, 342.08, 27.85, 73.14, 171.52, 222.54, 296.72, 243.58, 119.81, 297.17, 21.02, 247.54, 325.15, 60.93, 155.12, 288.79, 198.04, 199.76, 95.39, 287.11, 320.81, 227.73, 15.45 };
127 		double[] C = { 1934.136, 32964.467, 20.186, 445267.112, 45036.886, 22518.443, 65928.934, 3034.906, 9037.513, 33718.147, 150.678, 2281.226, 29929.562, 31555.956, 4443.417, 67555.328, 4562.452, 62894.029, 31436.921, 14577.848, 31931.756, 34777.259, 1222.114, 16859.074 };
128 
129 		for (int i = 0; i < 24; i++) {
130 			result += A[i] * specialCos(B[i] + (C[i] * T));
131 		}
132 		return result;
133 	}

135 	/**
136 	 * Julian Date to UTC Date Object.<br>
137 	 * <br>
138 	 * <i>See Meeus Astronmical Algorithms Chapter 7</i>
139 	 * 
140 	 * @param JDE = Julian date, possible with fractional days.
141 	 * @return Date = Output is a UTC date object.
142 	 */
143 	private static Date fromJDEToUTC(double JDE) {
144 		Date result = null;
145 		int A = 0;
146 		int alpha = 0;
147 
148 		// Integer (important) JDE's
149 		int Z = new Double(Math.floor(JDE + 0.5)).intValue();
150 		// Fractional JDE's
151 		double F = (JDE + 0.5) - Z;
152 
153 		if (Z < 2299161) {
154 			A = Z;
155 		} else {
156 			alpha = new Double(Math.floor((Z - 1867216.25) / 36524.25)).intValue();
157 			A = Z + 1 + alpha - new Double(Math.floor(alpha / 4)).intValue();
158 		}
159 		int B = A + 1524;
160 		int C = new Double(Math.floor((B - 122.1) / 365.25)).intValue();
161 		int D = new Double(Math.floor(365.25 * C)).intValue();
162 		int E = new Double(Math.floor((B - D) / 30.6001)).intValue();
163 
164 		// Day of month with decimals for time
165 		double DT = B - D - new Double(Math.floor(30.6001 * E)).intValue() + F;
166 
167 		// Month Number
168 		int Mon = E - ((E < 13.5) ? 1 : 13);
169 
170 		// Year 
171 		int Yr = C - (((double) Mon > 2.5) ? 4716 : 4715);
172 
173 		// Day of Month without decimals for time
174 		int Day = new Double(Math.floor(DT)).intValue();
175 
176 		// Hours and fractional hours
177 		double H = 24 * (DT - Day);
178 
179 		// Integer value of hours
180 		int Hr = new Double(Math.floor(H)).intValue();
181 
182 		// Minutes and fractional minutes
183 		double M = 60 * (H - Hr);
184 
185 		// Integer value of minutes
186 		int Min = new Double(Math.floor(M)).intValue();
187 
188 		// Integer value of seconds (Milliseconds discarded/cut away)
189 		int Sec = new Double(Math.floor(60 * (M - Min))).intValue();
190 
191 		Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
192 		cal.set(Calendar.YEAR, Yr);
193 		cal.set(Calendar.MONTH, Mon - 1);
194 		cal.set(Calendar.DAY_OF_MONTH, Day);
195 		cal.set(Calendar.HOUR_OF_DAY, Hr);
196 		cal.set(Calendar.MINUTE, Min);
197 		cal.set(Calendar.SECOND, Sec);
198 		result = cal.getTime();
199 
200 		return result;
201 	}

203 	/**
204 	 * Correct TDT to UTC.<br>
205 	 * <br>
206 	 * <i>See Meeus Astronmical Algorithms Chapter 10</i>
207 	 * 
208 	 * @param TDT
209 	 * @return
210 	 * @throws Exception
211 	 */
212 	private static Date fromTDTToUTC(Date TDT) throws Exception {
213 		Date result = null;
214 		// Range of years in lookup table.
215 		int tblFirstValue = 1620;
216 		int tblLastValue = 2002;
217 
218 		// @formatter:off
219 		// Correction lookup table has entry for every even year between tblFirstValue and tblLastValue.
220 		// Corrections in Seconds!
221 		double[] table = {
222 		/*1620*/121, 112, 103, 95, 88, 82, 77, 72, 68, 63, 60, 56, 53, 51, 48, 46, 44, 42, 40, 38,
223 		/*1660*/35, 33, 31, 29, 26, 24, 22, 20, 18, 16, 14, 12, 11, 10, 9, 8, 7, 7, 7, 7,
224 		/*1700*/7, 7, 8, 8, 9, 9, 9, 9, 9, 10, 10, 10, 10, 10, 10, 10, 10, 11, 11, 11,
225 		/*1740*/11, 11, 12, 12, 12, 12, 13, 13, 13, 14, 14, 14, 14, 15, 15, 15, 15, 15, 16, 16,
226 		/*1780*/16, 16, 16, 16, 16, 16, 15, 15, 14, 13,
227 		/*1800*/13.1, 12.5, 12.2, 12.0, 12.0, 12.0, 12.0, 12.0, 12.0, 11.9, 11.6, 11.0, 10.2, 9.2, 8.2,
228 		/*1830*/7.1, 6.2, 5.6, 5.4, 5.3, 5.4, 5.6, 5.9, 6.2, 6.5, 6.8, 7.1, 7.3, 7.5, 7.6,
229 		/*1860*/7.7, 7.3, 6.2, 5.2, 2.7, 1.4, -1.2, -2.8, -3.8, -4.8, -5.5, -5.3, -5.6, -5.7, -5.9,
230 		/*1890*/-6.0, -6.3, -6.5, -6.2, -4.7, -2.8, -0.1, 2.6, 5.3, 7.7, 10.4, 13.3, 16.0, 18.2, 20.2,
231 		/*1920*/21.1, 22.4, 23.5, 23.8, 24.3, 24.0, 23.9, 23.9, 23.7, 24.0, 24.3, 25.3, 26.2, 27.3, 28.2,
232 		/*1950*/29.1, 30.0, 30.7, 31.4, 32.2, 33.1, 34.0, 35.0, 36.5, 38.3, 40.2, 42.2, 44.5, 46.5, 48.5,
233 		/*1980*/50.5, 52.5, 53.8, 54.9, 55.8, 56.9, 58.3, 60.0, 61.6, 63.0, 63.8, 64.3 };/*2002 last entry*/
234 		// @formatter:on
235 
236 		/*
237 		 *  Values for deltaT for 2000 thru 2002 from NASA.
238 		 */
239 
240 		// deltaT = TDT - UTC (in Seconds).
241 		double deltaT = 0.0;
242 		Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("UTC"));
243 		cal.setTime(TDT);
244 		int Year = cal.get(Calendar.YEAR);
245 
246 		// Centuries from the epoch 2000.0
247 		double t = (Year - 2000.0) / 100.0;
248 
249 		/*
250 		 *  Find correction in table.
251 		 */
252 		if (Year >= tblFirstValue && Year <= tblLastValue) {
253 			// Odd year - interpolate.
254 			if (Year % 2 != 0) {
255 				deltaT = table[(Year - tblFirstValue - 1) / 2] + table[(Year - tblLastValue + 1 / 2)] / 2;
256 			} else {
257 				// Even year - direct table lookup.
258 				deltaT = table[(Year - tblFirstValue) / 2];
259 			}
260 		} else if (Year < 948) {
261 			deltaT = 2177 + 497 * t + 44.1 * Math.pow(t, 2);
262 		} else if (Year >= 948) {
263 			deltaT = 102.0 + 102.0 * t + 25.3 * Math.pow(t, 2);
264 			// Special correction to avoid discontinuity in 2000.
265 			if (Year >= 2000 && Year <= 2100) {
266 				deltaT += 0.37 * (Year - 2100);
267 			}
268 		} else {
269 			throw new Exception("Error: TDT to UTC correction not computed.");
270 		}
271 		cal.setTimeInMillis(TDT.getTime() - (long) (deltaT * 1000));
272 		result = cal.getTime();
273 
274 		return result;
275 	}
